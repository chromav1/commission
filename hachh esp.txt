-- hatch esp
local plr = game.Players.LocalPlayer
local hatches = {}

local function makeESP(trigger) 
    -- main 
    local label = Drawing.new("Text") 
    label.Size = 16
    label.Center = true 
    label.Outline = true
    label.Visible = false
    label.Color = Color3.fromRGB(255, 255, 0) -- yellow
    label.Font = 2
    
    -- distance  
    local distLabel = Drawing.new("Text")
    distLabel.Size = 14
    distLabel.Center = true
    distLabel.Outline = true 
    distLabel.Visible = false
    distLabel.Color = Color3.fromRGB(255, 255, 255)
    distLabel.Font = 2
    
    hatches[trigger] = {
        label = label,
        distLabel = distLabel
    }
end

local function updateESP()
    local char = plr.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    
    for trigger, esp in pairs(hatches) do 
        if trigger and trigger.Parent then
            -- put the esp above the hatch
            local pos = trigger.Position + Vector3.new(0, 3, 0)
            local screenPos, onScreen = WorldToScreen(pos)
            
            if onScreen then
                esp.label.Text = "Hatch"
                esp.label.Position = Vector2.new(screenPos.X, screenPos.Y - 10)
                esp.label.Visible = true
                
                if root then
                    -- calc farness
                    local dx = root.Position.X - trigger.Position.X
                    local dy = root.Position.Y - trigger.Position.Y
                    local dz = root.Position.Z - trigger.Position.Z
                    local dist = math.sqrt(dx*dx + dy*dy + dz*dz)
                    esp.distLabel.Text = math.floor(dist) .. "m"
                    esp.distLabel.Position = Vector2.new(screenPos.X, screenPos.Y + 10)
                    esp.distLabel.Visible = true
                else
                    esp.distLabel.Visible = false
                end 
            else
                -- hide
                esp.label.Visible = false
                esp.distLabel.Visible = false
            end
        else
            -- esp go slep
            esp.label:Remove()
            esp.distLabel:Remove()
            hatches[trigger] = nil
        end
    end
end 

local function scanForHatches()
    local map = workspace:FindFirstChild("Map")
    if not map then
        return 0
    end
    
    local newFound = 0
    
    -- lf hact wid da triganiga
    for _, thing in ipairs(map:GetDescendants()) do
        if thing:IsA("Model") and thing.Name == "Hatch" then
            -- find the trigger par
            local trigger = thing:FindFirstChild("Trigger")
            if trigger and trigger:IsA("BasePart") then
                -- add esp squared
                if not hatches[trigger] then
                    makeESP(trigger) 
                    newFound = newFound + 1
                    print("found new hatch at:", thing:GetFullName())
                end
            end
        end
    end
    
    return newFound
end

-- wait for map to load
wait(2)
print("starting hatch esp") 
local initialCount = scanForHatches()
print("initial scan complete - tracking " .. initialCount .. " hatches")

-- 15 20 years dagestan and forget
spawn(function()
    while true do
        wait(math.random(15, 20))
        local found = scanForHatches()
        if found > 0 then
            print("rescan found " .. found .. " new hatches")
        end
    end
end)

-- update cubed
while true do
    if isrbxactive() then
        updateESP()
    end
    wait(0.01)
end
