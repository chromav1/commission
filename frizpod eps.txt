-- freezepod esp 
local plr = game.Players.LocalPlayer
local pods = {}

local function makeESP(trigger)
    -- main 
    local label = Drawing.new("Text")
    label.Size = 16
    label.Center = true
    label.Outline = true
    label.Visible = false
    label.Color = Color3.fromRGB(0, 191, 255)
    label.Font = 2
    
    -- distance text 
    local distLabel = Drawing.new("Text")
    distLabel.Size = 14
    distLabel.Center = true
    distLabel.Outline = true
    distLabel.Visible = false
    distLabel.Color = Color3.fromRGB(255, 255, 255)
    distLabel.Font = 2
    
    pods[trigger] = {
        label = label,
        distLabel = distLabel
    }
end

local function updateESP()
    local char = plr.Character
    local root = char and char:FindFirstChild("HumanoidRootPart")
    
    for trigger, esp in pairs(pods) do
        if trigger and trigger.Parent then
            -- put the esp above
            local pos = trigger.Position + Vector3.new(0, 3, 0)
            local screenPos, onScreen = WorldToScreen(pos)
            
            if onScreen then
                esp.label.Text = "FreezePod"
                esp.label.Position = Vector2.new(screenPos.X, screenPos.Y - 10)
                esp.label.Visible = true
                
                if root then
                    -- calc how far
                    local dx = root.Position.X - trigger.Position.X
                    local dy = root.Position.Y - trigger.Position.Y
                    local dz = root.Position.Z - trigger.Position.Z
                    local dist = math.sqrt(dx*dx + dy*dy + dz*dz)
                    esp.distLabel.Text = math.floor(dist) .. "m"
                    esp.distLabel.Position = Vector2.new(screenPos.X, screenPos.Y + 10)
                    esp.distLabel.Visible = true
                else
                    esp.distLabel.Visible = false
                end
            else
                -- not visible
                esp.label.Visible = false
                esp.distLabel.Visible = false
            end
        else
            -- pod is gon
            esp.label:Remove()
            esp.distLabel:Remove()
            pods[trigger] = nil
        end
    end
end

local function scanForPods()
    local map = workspace:FindFirstChild("Map")
    if not map then
        return 0
    end
    
    local newFound = 0
    
    -- look for podtrigger 
    for _, thing in ipairs(map:GetDescendants()) do
        if thing:IsA("BasePart") and thing.Name == "PodTrigger" then
            -- add esp again I think I forgot
            if not pods[thing] then
                makeESP(thing)
                newFound = newFound + 1
                print("found new freezepod at:", thing:GetFullName())
            end
        end
    end
    
    return newFound
end

-- wait for map to load then scan
wait(2)
print("starting freezepod esp")
local initialCount = scanForPods()
print("initial scan complete - tracking " .. initialCount .. " freezepods")

-- check 15 20
spawn(function()
    while true do
        wait(math.random(15, 20))
        local found = scanForPods()
        if found > 0 then
            print("rescan found " .. found .. " new freezepods")
        end
    end
end)

-- updatingnation
while true do
    if isrbxactive() then
        updateESP()
    end
    wait(0.01)
end
